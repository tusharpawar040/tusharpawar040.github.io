#include <iostream>
using namespace std;

class Date
{
private:
    int day;
    int month;
    int year;

public:
    Date(int d, int m, int y)
    {
        day = d;
        month = m;
        year = y;
    }

    void setDay(int d)
    {
        day = d;
    }
    void setMonth(int m)
    {
        month = m;
    }
    void setYear(int y)
    {
        year = y;
    }
    int getDay()
    {
        return day;
    }
    int getMonth()
    {
        return month;
    }
    int getYear()
    {
        return year;
    }

    void display()
    {
        cout << day << "/" << month << "/" << year << endl;
    }

    bool isLeapYear()
    {

        if (year % 4 == 0 || year % 400 == 0)
        {
            return 1;
        }

        else
        {
            return 0;
        }
    }

    {
        if (d <= 27)
        {
            d++;
        }
        else if (d == 28)
        {
            if (m == 2)
            {
                if (isLeapYear())
                {
                    d++;
                }
                else
                {
                    d = 1;
                    m++;
                }
            }
            
            else
            {
                d++;
            }
        }
        else if (d == 29)
        {
        	
            d = 1;
            m++;
        }
        else if (d == 30)
        {
            if ((m == 4) || (m == 6) || (m == 9) || (m == 11))
            {
                d = 1;
                m++;
            }
            else
            {
                d++;
            }
        }
        
        else
        {
            if (m == 12)
            {
            	
                d = 1;
                m = 1;
                y++;
            }
           
            else
            {
                d = 1;
                m++;
            }
        }
        cout << "Incremented Date is\n";
        cout << d << "/" << m << "/" << y << endl;
    }
};
